# =============================================================
#  一、Python 包依赖 (pip install)
# =============================================================
#
#  这些包用于平台后端的代码沙盒执行服务 (code-runner.js)。
#  学生提交的 Python 实验代码会在沙盒中自动注入 python_helper.py，

# ----- 核心科学计算 -----
# 导入方式: import numpy as np
# 使用位置: python_helper.py (自动注入所有实验); 全部 74 个实验模板
# 用途: 数组/矩阵运算、FFT、数据类型转换、像素级操作
numpy>=1.24.0

# ----- 计算机视觉 (OpenCV) -----
# 导入方式: import cv2
# 使用位置: python_helper.py (自动注入); 全部实验模板
# 用途: 图像读写、灰度化、滤波(高斯/中值/双边)、边缘检测(Canny/Sobel)、
#       形态学操作(腐蚀/膨胀/开闭)、阈值分割、仿射变换、直方图均衡、
#       霍夫变换、轮廓检测、色彩空间转换
# 说明: headless 版本不含 GUI 组件，适合服务端运行
#       本地开发如需 imshow 等 GUI 功能，可替换为 opencv-python
opencv-python-headless>=4.8.0

# ----- 图像处理 (Pillow / PIL) -----
# 导入方式: from PIL import Image, ImageFilter, ImageEnhance, ImageOps, ImageChops
# 使用位置: python_helper.py (自动注入); ExperimentLab 实验模板; case-compression 实验
# 用途: 图像类型转换 (ndarray <-> PIL Image)、图像增强、滤波、压缩模拟、
#       JPEG 编码、位平面操作、图像合并与通道操作
Pillow>=10.0.0

# ----- 数据可视化 -----
# 导入方式: import matplotlib; matplotlib.use('Agg'); import matplotlib.pyplot as plt
# 使用位置: python_helper.py (自动注入，设置 Agg 非交互后端); 大部分实验模板
# 用途: 直方图绘制、频谱图、对比图、多子图排列、结果保存为 PNG
# 环境变量: MPLBACKEND=Agg (必须设置，否则在无显示器环境下报错)
matplotlib>=3.7.0

# ----- 科学计算与信号处理 -----
# 导入方式: from scipy import ndimage; from scipy.fftpack import dct, idct
# 使用位置: ExperimentLab 实验 2 (中值滤波/Sobel), 实验 3 (DCT/IDCT),
#          实验 4 (形态学), 实验 5 (边缘检测)
# 用途: ndimage 模块 — 中值滤波、Sobel 边缘检测、形态学操作、卷积
#       fftpack 模块 — 离散余弦变换 (DCT) 及其逆变换 (IDCT)
scipy>=1.11.0

# ----- 高级图像处理 (scikit-image) -----
# 导入方式: from skimage.morphology import skeletonize
# 使用位置: code-templates.js 中的 skeleton 实验模板 (骨架化/Zhang-Suen 算法)
# 用途: 骨架化提取、区域标注、高级形态学处理
scikit-image>=0.21.0

# ----- PPT 文本提取工具 (辅助脚本，非平台运行时依赖) -----
# 导入方式: import olefile
# 使用位置: read_ppt.py, read_ppt2.py, extract_all_ppts.py
# 用途: 读取旧版 .ppt 格式 (OLE2 复合文档) 中的文本内容
# 说明: 这三个脚本是独立的数据提取工具，不参与平台运行
#       如不需要提取 PPT 可不安装
olefile>=0.46


# =============================================================
#  二、系统级依赖 (apt 包 / Docker 环境)
# =============================================================
#
#  以下依赖通过 apt-get 安装 (Docker 基础镜像: node:18-bookworm)。
#  Windows 本地开发时大部分不需要安装。
#
#  ---- Python 运行时 ----
#  python3          — Python 3 解释器 (代码沙盒执行引擎)
#  python3-pip      — pip 包管理器
#  python3-venv     — 虚拟环境支持
#
#  ---- GNU Octave (MATLAB 兼容引擎) ----
#  octave           — Octave 主程序 (提供 octave-cli 命令行执行)
#  octave-image     — Octave 图像处理工具箱
#  octave-io        — Octave I/O 包 (Excel/CSV 读写)
#  说明: Octave 用于执行 MATLAB 风格的实验代码 (code-runner.js 中调用)
#        环境变量 OCTAVE_PATH=/usr/bin/octave-cli
#
#  ---- OpenCV 运行时库 ----
#  libgl1           — OpenGL 渲染库 (cv2 初始化必需)
#  libglib2.0-0     — GLib 运行时 (cv2 内部依赖)
#  libsm6           — X11 Session Management (cv2 headless 依赖)
#  libxext6         — X11 扩展库
#  libxrender1      — X11 渲染库
#  说明: 缺少以上任何一个库，import cv2 都会报 libGL/libglib 找不到错误
#
#  ---- 字体 ----
#  fonts-wqy-zenhei — 文泉驿正黑字体
#  说明: matplotlib 绘制含中文标题的图表时必需，否则中文显示为方块
#
#  ---- Docker 安装命令 (已集成在 Dockerfile 中) ----
#  apt-get update && apt-get install -y --no-install-recommends \
#      python3 python3-pip python3-venv \
#      octave octave-image octave-io \
#      libgl1 libglib2.0-0 libsm6 libxext6 libxrender1 \
#      fonts-wqy-zenhei \
#      && rm -rf /var/lib/apt/lists/*
#
#  ---- Windows 本地开发说明 ----
#  - Python: 从 https://www.python.org 下载安装 Python 3.9+
#  - Node.js: 从 https://nodejs.org 下载安装 Node.js 18+
#  - Octave: 从 https://octave.org/download 下载安装 (可选，仅运行 Octave 实验时需要)
#  - 中文字体: Windows 自带微软雅黑，matplotlib 可自动识别
#  - OpenCV 系统库: Windows 版 pip 包已自带 DLL，无需额外安装
